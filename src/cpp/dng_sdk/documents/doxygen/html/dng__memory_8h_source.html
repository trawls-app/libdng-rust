310 
311  uint32 PhysicalSize ()
312  {
313 
314  // This size is padded for TWO reasons! The first is allow
315  // alignment to 16-byte boundaries if the allocator does not do
316  // that already. The second, which is very important, so to
317  // provide safe overread areas for SSE2-type bottlenecks, which
318  // can often be written faster by allowing them to reading
319  // slightly block. Someone on the image core them did not
320  // understand this and removed this padding. I'm undoing this
321  // removal and restoring this padding, since removing it might
322  // lead to memory access crashes in some cases.
323  //
324  // Please do NOT change the following padding unless you are very
325  // sure what you are doing.
326 
327  dng_safe_uint32 safeLogicalSize (fLogicalSize);
328 
329  #if qDNGAVXSupport
330 
331  // Allow 32-byte alignment + 64-byte overread in both directions:
332  // 32 + 64 + 64 = 160.
333 
334  return (safeLogicalSize + 160u).Get ();
335 
336  #else
337 
338  // Allow 16-byte alignment + overread.
339 
340  return (safeLogicalSize + 64u).Get ();
341 
342  #endif // qDNGAVXSupport
343 
344  }

603  T * allocate (size_t n)
604  {
605  const size_t size = SafeSizetMult (n, sizeof (T));
606  T *retval = static_cast<T *> (malloc (size));
607  if (!retval)
608  {
609  ThrowMemoryFull ();
610  }
611  return retval;
612  }
