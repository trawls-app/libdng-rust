196  // Color matrices for up to two calibrations.
197 
198  // These matrices map XYZ values to non-white balanced camera values.
199  // Adobe needs to go that direction in order to determine the clipping
200  // points for highlight recovery logic based on the white point. If
201  // cameras were all 3-color, the matrix could be stored as a forward matrix,
202  // but we need the backwards matrix to deal with 4-color cameras.
203 
204  dng_matrix fColorMatrix1;
205  dng_matrix fColorMatrix2;
206 
207  // These matrices map white balanced camera values to XYZ chromatically
208  // adapted to D50 (the ICC profile PCS white point). If the matrices
209  // exist, then this implies that white balancing should be done by scaling
210  // camera values with a diagonal matrix.
211 
212  dng_matrix fForwardMatrix1;
213  dng_matrix fForwardMatrix2;
214 
215  // Dimensionality reduction hints for more than three color cameras.
216  // This is an optional matrix that maps the camera's color components
217  // to 3 components. These are only used if the forward matrices don't
218  // exist, and are used invert the color matrices.
219 
220  dng_matrix fReductionMatrix1;
221  dng_matrix fReductionMatrix2;
